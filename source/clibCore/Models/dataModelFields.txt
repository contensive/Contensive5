function m
dim cs,c,n
set cs = cp.csNew()
if ( cs.open( "content" )) then
  do
    n = replace( cs.getText( "name" ), " ", "")
	if ( right( n,1 ) = "s" ) then
		n = left( n, len( n )-1)
	end if
	n = n & "Model"
    c = c & vbcrlf & vbcrlf & "---------------------------------------------"
    c = c & vbcrlf & n
    c = c & vbcrlf & getModelFields( cs.getInteger( "id" ))
    cs.goNext()
  loop while cs.ok
end if
cs.close()
call cp.file.saveVirtual( "Models.txt", c )
m = "<a href=""" & cp.site.filePath & "Models.txt"">Models.txt</a>"
end function

function getModelFields( contentId )
dim cs, t, p, l, s
set cs = cp.csNew()
if ( cs.open( "content fields", "contentid=" & contentid, "name" )) then
  do
    select case cs.getInteger( "type" )
      case 13: 
		' autoincrement (identity)
		p = vbcrlf & "public " & cs.getText( "name" ) & " as Integer" & p
		l = vbcrlf & "." & cs.getText( "name" ) & " = cs.getInteger(""" & cs.getText( "name" ) & """)" & l
		s = s & ""
      case 1,7,15: 
		p = p & vbcrlf & "public " & cs.getText( "name" ) & " as Integer"
		l = l & vbcrlf & "." & cs.getText( "name" ) & " = cs.getInteger(""" & cs.getText( "name" ) & """)"
		s = s & vbcrlf & "cs.SetField(""" & cs.getText( "name" ) & """, " & cs.getText( "name" ) & ".ToString())"
      case 4:
		p = p & vbcrlf & "public " & cs.getText( "name" ) & " as Boolean"
		l = l & vbcrlf & "." & cs.getText( "name" ) & " = cs.getBoolean(""" & cs.getText( "name" ) & """)"
		s = s & vbcrlf & "cs.SetField(""" & cs.getText( "name" ) & """, " & cs.getText( "name" ) & ".ToString())"
      case 2,3,6,10,11,16,17,18,19,20,21,22:
		p = p & vbcrlf & "public " & cs.getText( "name" ) & " as String"
		l = l & vbcrlf & "." & cs.getText( "name" ) & " = cs.getText(""" & cs.getText( "name" ) & """)"
		s = s & vbcrlf & "cs.SetField(""" & cs.getText( "name" ) & """, " & cs.getText( "name" ) & ")"
      case 5:
		p = p & vbcrlf & "public " & cs.getText( "name" ) & " as Date"
		l = l & vbcrlf & "." & cs.getText( "name" ) & " = cs.getDate(""" & cs.getText( "name" ) & """)"
		s = s & vbcrlf & "cs.SetField(""" & cs.getText( "name" ) & """, " & cs.getText( "name" ) & ".ToString())"
      case 8,14:
	    ' ignore
      case 9,12:
		p = p & vbcrlf & "public " & cs.getText( "name" ) & " as Double"
		l = l & vbcrlf & "." & cs.getText( "name" ) & " = cs.getNumber(""" & cs.getText( "name" ) & """)"
		s = s & vbcrlf & "cs.SetField(""" & cs.getText( "name" ) & """, " & cs.getText( "name" ) & ".ToString())"
    end select
    cs.goNext()
  loop while cs.ok
end if
cs.close()
getModelFields = p & vbcrlf & l & vbcrlf & s
end function
